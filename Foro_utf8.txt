          author: "Prof. García",

      function applyForumRole() {
        const userRole = document.getElementById("userRole");
        const userName = document.getElementById("userName");
        const adminPanel = document.getElementById("adminPanel");
        if (isTeacher) {
          if (userRole) userRole.textContent = "Docente";
          if (userName) userName.textContent = "Prof. García";
          if (adminPanel) adminPanel.style.display = "block";
        } else {
          if (userRole) userRole.textContent = "Estudiante";
          if (userName) userName.textContent = "María González";
          if (adminPanel) adminPanel.style.display = "none";
        }
      }

      function toggleRole() {
        isTeacher = !isTeacher;
        try { localStorage.setItem('qs_role', isTeacher ? 'docente' : 'estudiante'); } catch(e) {}
        applyForumRole();
      }

      function createTopic() {
        const title = document.getElementById("topicTitle").value;
        const category = document.getElementById("topicCategory").value;
        const content = document.getElementById("topicContent").value;

        if (!title || !content) {
          alert("Por favor completa todos los campos");
          return;
        }

        // Simular creación de tema
        alert("✅ Tema creado exitosamente: " + title);

        // Limpiar formulario
        document.getElementById("topicTitle").value = "";
        document.getElementById("topicContent").value = "";
      }

      function openTopic(topicId) {
        currentTopicId = topicId;
        const topic = topics[topicId];

        document.getElementById("modalTitle").textContent = topic.title;
        document.getElementById("modalCategory").textContent = topic.category;
        document.getElementById("modalContent").textContent = topic.content;

        // Mostrar respuestas
        const responsesList = document.getElementById("responsesList");
        responsesList.innerHTML = "";

        topic.responses.forEach((response) => {
          const responseDiv = document.createElement("div");
          responseDiv.className = "border-l-4 border-green-400 pl-4 py-2";
          responseDiv.innerHTML = `
                    <div class="flex items-center space-x-2 mb-2">
                        <span class="font-medium text-gray-800">${response.author}</span>
                        <span class="text-xs text-gray-500">${response.time}</span>
                    </div>
                    <p class="text-gray-700">${response.text}</p>
                `;
          responsesList.appendChild(responseDiv);
        });

        document.getElementById("topicModal").classList.remove("hidden");
      }

      function closeTopic() {
        document.getElementById("topicModal").classList.add("hidden");
        document.getElementById("responseText").value = "";
      }

      function addResponse() {
        const responseText = document.getElementById("responseText").value;

        if (!responseText.trim()) {
          alert("Por favor escribe una respuesta");
          return;
        }

        const userName = isTeacher ? "👨‍🏫 Prof. García" : "👩‍🎓 María González";

        // Agregar respuesta al tema
        topics[currentTopicId].responses.push({
          author: userName,
          text: responseText,
          time: "Ahora",
        });

        // Actualizar vista
        openTopic(currentTopicId);

        alert("✅ Respuesta enviada correctamente");
      }

      // Cerrar modal al hacer clic fuera
      document
        .getElementById("topicModal")
        .addEventListener("click", function (e) {
          if (e.target === this) {
            closeTopic();
          }
        });
    </script>
