🎯 Demo: Automatización de Login

1️⃣ Configuración del entorno
2️⃣ Localización de elementos
3️⃣ Interacción con formularios
4️⃣ Validación de resultados

Ejemplo: Selenium WebDriver

// Selenium WebDriver - Java
import org.openqa.selenium.*;
import org.openqa.selenium.chrome.*;
public class LoginTest {
WebDriver driver = new ChromeDriver();
@Test
public void testLogin() {
driver.get("https://example.com/login");
driver.findElement(By.id("username"))
.sendKeys("usuario@test.com");
driver.findElement(By.id("password"))
.sendKeys("password123");
driver.findElement(By.id("loginBtn"))
.click();
// Validar resultado
String welcomeMsg = driver
.findElement(By.id("welcome"))
.getText();
assert(welcomeMsg.contains("Bienvenido"));
}
}
